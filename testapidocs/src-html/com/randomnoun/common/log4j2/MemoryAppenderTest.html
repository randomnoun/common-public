046        public void testMemoryAppenderViaProperties_Log2Logger() {
047
048                Properties props = new Properties();
049                props.put("log4j.rootCategory", "INFO, MEMORY");
050                props.put("log4j.appender.MEMORY", "com.randomnoun.common.log4j2.MemoryAppender");
051                props.put("log4j.appender.MEMORY.MaximumLogSize", 100);
052                
053                // layouts have no effect on MemoryAppenders
054                //props.put("log4j.appender.MEMORY.layout", "org.apache.log4j.PatternLayout");
055                //props.put("log4j.appender.MEMORY.layout.ConversionPattern", "%d{dd/MM/yy HH:mm:ss.SSS} %-5p %c - %m%n");
056                // PropertyConfigurator.configure(props);
057                
058                Log4j2CliConfiguration lcc = new Log4j2CliConfiguration();
059                lcc.init("[MemoryAppenderTest]", props);
060                
061                Log4j1ConfigurationParser lcp = new Log4j1ConfigurationParser();
062                ConfigurationBuilder<?> builder = lcp.buildConfigurationBuilder(props);
063                LoggerContext ctx = Configurator.initialize(builder.build());
064
065                org.apache.logging.log4j.Logger logger = LogManager.getLogger("testLogger");
066                MemoryAppender memoryAppender = ctx.getConfiguration().getAppender("MEMORY");
067                
068                List<LogEvent> logEvents;
069
070                long start = System.currentTimeMillis();
071                logger.info("info message");
072                long end = System.currentTimeMillis();
073                
074                logEvents = memoryAppender.getLogEvents();
075                assertTrue("info message in memoryAppender", logEvents.size()==1);
076                assertEquals("info message", logEvents.get(0).getMessage().getFormattedMessage());
077                // assertEquals("info message", logEvents.get(0).getRenderedMessage());
078                assertNotNull(logEvents.get(0).getLoggerName());
079                assertEquals(Level.INFO, logEvents.get(0).getLevel());
080                // NB: timestamp field requires log4j 1.2.15
081                assertTrue("timestamp of loggingEvent >= start", logEvents.get(0).getTimeMillis() >= start);
082                assertTrue("timestamp of loggingEvent <= end", logEvents.get(0).getTimeMillis() <= end);
083                
084                logger.debug("debug message");
085                logEvents = memoryAppender.getLogEvents();
086                assertTrue("debug message suppressed from memoryAppender", logEvents.size()==1);
087                
088        }

133        public void testLogWindowSize() {
134
135                // create a MemoryLogger capable of holding 10 entries
136                
137                Properties props = new Properties();
138                props.put("log4j.rootCategory", "INFO, MEMORY");
139                props.put("log4j.appender.MEMORY", "com.randomnoun.common.log4j2.MemoryAppender");
140                props.put("log4j.appender.MEMORY.MaximumLogSize", "10");
141                // PropertyConfigurator.configure(props);
142                Log4j2CliConfiguration lcc = new Log4j2CliConfiguration();
143                lcc.init("[MemoryAppenderTest]", props);
144                
145                Log4j1ConfigurationParser lcp = new Log4j1ConfigurationParser();
146                ConfigurationBuilder<?> builder = lcp.buildConfigurationBuilder(props);
147                LoggerContext ctx = Configurator.initialize(builder.build());
148
149                Logger logger = Logger.getLogger("testLogger");
150                // MemoryAppender memoryAppender = (MemoryAppender) Logger.getRootLogger().getAppender("MEMORY");
151                MemoryAppender memoryAppender = ctx.getConfiguration().getAppender("MEMORY");
152                
153                List<LogEvent> logEvents;
154
155                // write five messages. 
156                // the 0th message in MemoryAppender should be the most recent (i.e. message #5)
157                for (int i=1; i<=5; i++) {
158                        logger.info("message number " + i);
159                }
160                logEvents = memoryAppender.getLogEvents();
161                assertTrue("5 info messages in memoryAppender", logEvents.size()==5);
162                for (int i=0; i<5; i++) {
163                        assertEquals("message number " + (5 - i), logEvents.get(i).getMessage().getFormattedMessage());
164                }
165                
166                // write another five messages
167                for (int i=6; i<=10; i++) {
168                        logger.info("message number " + i);
169                }
170                logEvents = memoryAppender.getLogEvents();
171                assertTrue("10 info messages in memoryAppender", logEvents.size()==10);
172                for (int i=0; i<10; i++) {
173                        assertEquals("message number " + (10 - i), logEvents.get(i).getMessage().getFormattedMessage());
174                }
175                
176                // write another five messages (memoryAppender should just contain messages 6-15)
177                for (int i=11; i<=15; i++) {
178                        logger.info("message number " + i);
179                }
180                logEvents = memoryAppender.getLogEvents();
181                //System.out.println(loggingEvents.size());
182                assertTrue("10 info messages in memoryAppender", logEvents.size()==10);
183                for (int i=0; i<10; i++) {
184                        assertEquals("message number " + (15 - i), logEvents.get(i).getMessage().getFormattedMessage());
185                }
186                
187        }
