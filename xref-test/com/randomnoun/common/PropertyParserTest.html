21      public void testSimpleProps() throws ParseException, IOException {
22      	String in = 
23      		"a=b\n" +
24      		"# regular comment with an = in it\n" +
25      		"c=d";
26      			
27      	Reader r = new StringReader(in);
28      	PropertyParser pp = new PropertyParser(r);
29      	Properties p = pp.parse();
30      	
31          assertEquals(2, p.size());
32          assertEquals("b", p.getProperty("a"));
33          assertEquals("d", p.getProperty("c"));
34          
35          assertEquals(0, ((PropertiesWithComments) p).getComments().size());
36      }
37  
38      public void testPropsWithComments() throws ParseException, IOException {
39      	String in =
40      		"## comment for a\n" +
41      		"a=b\n" +
42      		"## comment for c\n" +
43      		"# that extends over multiple lines\n" +
44      		"c=d\n" +
45      		"e=f";
46      			
47      	Reader r = new StringReader(in);
48      	PropertyParser pp = new PropertyParser(r);
49      	PropertiesWithComments p = pp.parse();
50  
51          assertEquals(3, p.size());
52          assertEquals("b", p.getProperty("a"));
53          assertEquals("d", p.getProperty("c"));
54          assertEquals("f", p.getProperty("e"));
55  
56          assertEquals("comment for a", p.getComment("a"));
57          assertEquals("comment for c\nthat extends over multiple lines", p.getComment("c"));
58          assertEquals(null, p.getComment("e"));
59          assertEquals(2, p.getComments().size());
60  
61      }
