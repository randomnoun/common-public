096    /** Ensure that the database drivers are accessible, for simple and dbcp connection types 
097     * 
098     * @throws RuntimeExcption if the database drivers are not on the classpath of this webapp
099     */
100        protected void initDatabase() {
101        logger.info("Initialising database...");
102        try {
103                String connectionType = getProperty("database.connectionType");
104                if (connectionType==null || 
105                  connectionType.equals("simple") || 
106                  connectionType.equals("dbcp")) { 
107                        Class.forName(getProperty("database.driver")).getDeclaredConstructor().newInstance();
108                }
109        } catch (Exception e) {
110            initialisationFailure = e;
111            throw new RuntimeException("Could not load database driver '" + getProperty("database.driver") + "'", e);
112        }
113    }

115    /** Initialises log4j 
116     */
117    protected void initLogger() {
118        System.out.println("Initialising log4j...");
119        
120        Properties props = new Properties();
121        props.putAll(PropertyParser.restrict(this, "log4j", false));
122    
123        // replace ${xxxx}-style placeholders
124        String key, value, varName, varValue;
125        int startIdx, endIdx;
126        for (Enumeration<Object> e = props.keys(); e.hasMoreElements(); ) {
127                key = (String) e.nextElement();
128                value = (String) props.get(key);
129                startIdx = 0;
130                startIdx = value.indexOf("${", startIdx);
131                while (startIdx > -1) {
132                        endIdx = value.indexOf("}", startIdx+2);
133                        if (endIdx>-1) {
134                                varName = value.substring(startIdx+2, endIdx-startIdx-2);
135                                varValue = (String) this.get(varName);
136                                if (varValue == null) { varValue = ""; }
137                                value = value.substring(0, startIdx) + varValue + value.substring(endIdx + 1);
138                                startIdx = value.indexOf("${", startIdx); 
139                        } else {
140                                startIdx = -1;
141                        }
142                }
143        }
144        
145        PropertyConfigurator.configure(props);
146        logger.debug("log4j initialised");
147    }

149    protected void initSecurityContext() {
150        
151        if (securityContext==null) {
152                        logger.info("Initialising security context...");
153                        if (!"false".equals(getProperty("auth.enableSecurityContext"))) {
154                                Map<String, Object> securityProperties  = new HashMap<String, Object>();
155                                // no customerIds, applicationIds, auditUsernames or table suffixes ! Yay !
156                                String dbVendor = (String) this.get("database.vendor");
157                                if (Text.isBlank(dbVendor)) { dbVendor = SqlGenerator.DATABASE_MYSQL; }
158                                
159                                securityProperties.put(SpringSecurityLoaderImpl.INIT_DATABASE_VENDOR, dbVendor);
160                                securityProperties.put(SpringSecurityLoaderImpl.INIT_JDBCTEMPLATE, getJdbcTemplate());
161                                
162                                securityProperties.put(SecurityContext.INIT_CASE_INSENSITIVE, this.get("securityContext.caseInsensitive"));
163                                securityProperties.put(SecurityContext.INIT_USER_CACHE_SIZE, this.get("securityContext.userCacheSize"));
164                                securityProperties.put(SecurityContext.INIT_USER_CACHE_EXPIRY, this.get("securityContext.userCacheExpiry"));
165                                
166                                SecurityLoader securityLoader = new SpringSecurityLoaderImpl();
167                                SecurityAuthenticator securityAuthenticator = new SpringSecurityAuthenticatorImpl();
168                                
169                                securityContext = new SecurityContext(securityProperties, securityLoader, securityAuthenticator);
170                        } else {
171                                Map<String, Object> securityProperties  = new HashMap<String, Object>();
172                                SecurityLoader securityLoader = new NullSecurityLoaderImpl();
173                                SecurityAuthenticator securityAuthenticator = new NullSecurityAuthenticatorImpl();
174                                securityContext = new SecurityContext(securityProperties, securityLoader, securityAuthenticator);
175                        }
176        }
177    }

179    /** Sets the hostname application property. Should be called before {@link #loadProperties()},
180     * which may have ENVIRONMENT settings that rely on the hostname.
181     */
182    protected void initHostname() {
183        String hostname;
184        try {
185            java.net.InetAddress localMachine = java.net.InetAddress.getLocalHost();
186            hostname = localMachine.getHostName();
187        } catch(java.net.UnknownHostException uhe) {
188            hostname = "localhost";
189        }
190        this.put("hostname", hostname);
191    }

193    /** Returns a bundle to be used for internationalisation. The bundle will initially be
194     * searched for using the standard ResourceBundle.getBundle(name) method (i.e. in the
195     * server's classpath); and, if this fails, will return 
196     * ResourceBundle("resources.i18n." + name); which will search the i18n tree of the
197     * eomail project. 
198     * 
199     * @param i18nBundleName The bundle name to search for (e.g. "user")
200     * @param user The user for whom we are retrieving i18ned messages for (the locale of this
201     *   user will be used to determine which resource bundle to use)
202     * @return The ResourceBundle requested.
203     */    
204    public ResourceBundle getBundle(String i18nBundleName, User user) {
205        if (i18nBundleName.startsWith("resources")) {
206            return ResourceBundle.getBundle(i18nBundleName, user.getLocale());    
207        } else {
208            return ResourceBundle.getBundle("resources.i18n." + i18nBundleName, user.getLocale());
209        }
210        
211    }

214    /** Load the configuration from both the classpath and the
215     * filesystem. Properties defined on the filesystem will override any contained within
216     * the web application. 
217     * 
218     * <p>Multiple environments can be specified in properties files using the 
219     * STARTENVIRONMENT/ENDENVIRONMENT tags. The tags use the machine's hostname as an
220     * environment specifier. 
221     */
222    protected void loadProperties() {
223        String configPath = System.getProperty(getSystemPropertyKeyConfigPath());
224        if (configPath==null) { configPath = "."; }
225        File configFile = new File(configPath + getConfigResourceLocation());
226        try {
227            // load from resources
228            System.out.println("Loading properties file for environment '" + this.getProperty("hostname") + "'");
229            InputStream is = this.getClass().getClassLoader().getResourceAsStream(getConfigResourceLocation());
230            if (is==null && getConfigResourceLocation().charAt(0)=='/') {
231                // command-line support
232                is = this.getClass().getClassLoader().getResourceAsStream(getConfigResourceLocation().substring(1));
233            }
234            if (is==null) {
235                System.out.println("Could not find internal property file");
236            } else {
237                PropertyParser parser = new PropertyParser(new InputStreamReader(is), this.getProperty("hostname"));
238                Properties props = parser.parse();
239                is.close();
240                this.putAll(props);
241            }
242            
243            // then load from file (if it exists)
244            if (configFile.exists()) {
245                System.out.println("Loading properties file from '" + configFile.getCanonicalPath() + "'");
246                is = new BufferedInputStream(new FileInputStream(configFile));
247                PropertyParser parser = new PropertyParser(new InputStreamReader(is), this.getProperty("hostname"));
248                Properties props = parser.parse();
249                is.close();
250                this.putAll(props);
251            } else {
252                System.out.println("Properties file not found at '" + configFile.getCanonicalPath() + "' ... using defaults");
253                
254            }
255        } catch (Exception e) {
256            this.initialisationFailure = e;
257            throw new RuntimeException("Could not load " + getConfigResourceLocation(), e);
258        }
259    }

270    /** Returns a named jdbcTemplate object that can be used to query the database
271     *  
272     * @param connectionName The name of the connection
273     * 
274     * @return a jdbcTemplate to the database
275     */
276    public synchronized JdbcTemplate getJdbcTemplate(String connectionName) {
277        String prefix = connectionName == null ? "database." : "databases." + connectionName + ".";
278        
279        JdbcTemplate jt = jdbcTemplates.get(connectionName);
280        if (jt==null) {
281                String connectionType = getProperty(prefix + "connectionType");
282                if (connectionType==null) { connectionType = "simple"; }
283                String driver = getProperty(prefix + "driver");
284                String url = getProperty(prefix + "url");
285                String username = getProperty(prefix + "username");
286                String password = getProperty(prefix + "password");
287                logger.info("Retrieving " + connectionType + " connection for '" + username + "'/'" + password + "' @ '" + url + "'");
288                DataSource ds = null;
289                if (!Text.isBlank(driver)) { 
290                        try {
291                                Class.forName(driver); 
292                        } catch (ClassNotFoundException e) {
293                                        logger.error("Error loading class '" + driver + "' for database '" + connectionName + "'", e);
294                                } 
295                }
296                if (connectionType.equals("simple")) {
297                        if (url==null) { throw new NullPointerException(prefix + "url has not been initialised"); }
298                        Connection connection;
299                        try {
300                            if (Text.isBlank(username)) {
301                                connection = DriverManager.getConnection (url);
302                            } else {
303                                connection = DriverManager.getConnection (url, username, password);
304                            }
305                        } catch (SQLException sqle) {
306                            throw new DataAccessResourceFailureException("Could not open connection to database", sqle); 
307                        }
308                        ds = new SingleConnectionDataSource(connection, false);
309                        
310                } else if (connectionType.equals("dbcp")) {
311                        if (url==null) { throw new NullPointerException(prefix + "url has not been initialised"); }
312                        BasicDataSource bds = new BasicDataSource();
313                        bds.setDriverClassName(driver);
314                        Map<? extends Object, ? extends Object> dbcpProps = PropertyParser.restrict(this, prefix + "dbcp", true);
315                        Struct.setFromMap(bds, dbcpProps, false, true, false);
316                        bds.setUrl(url);
317                        if (!Text.isBlank(username)) {
318                                bds.setUsername(username);
319                                bds.setPassword(password);
320                        }
321                        ds = bds;
322                        
323                } else if (connectionType.equals("c3p0")) {
324                        if (url==null) { throw new NullPointerException(prefix + "url has not been initialised"); }
325                        ComboPooledDataSource  cpds = new ComboPooledDataSource ();
326                        try {
327                                        cpds.setDriverClass(driver);
328                                } catch (PropertyVetoException e) {
329                                        throw new IllegalStateException("Could not set driverClass '" + driver + "' for c3p0 datasource", e);
330                                }
331                        @SuppressWarnings("unchecked")
332                                Map<String,String> c3poProps = (Map<String, String>) PropertyParser.restrict(this, prefix + "c3p0", true);
333                        // remove extensions from props
334                        for (Iterator<Map.Entry<String,String>> i = c3poProps.entrySet().iterator(); i.hasNext(); ) { 
335                                Map.Entry<String,String> e=i.next(); 
336                                if (e.getKey().startsWith("extensions.")) { i.remove(); }
337                        }
338                        Struct.setFromMap(cpds, c3poProps, false, true, false);
339                        
340                        // set extensions separately
341                        @SuppressWarnings("unchecked")
342                                Map<String,String> c3poExtensionProps = (Map<String, String>) PropertyParser.restrict(this, prefix + "c3p0.extensions", true);
343                        cpds.setExtensions(c3poExtensionProps);
344                        cpds.setJdbcUrl(url);
345                        if (!Text.isBlank(username)) {
346                                cpds.setUser(username);
347                                cpds.setPassword(password);
348                        }
349                        ds = cpds;
350                        
351                } else if (connectionType.equals("jndi")) {
352                        String jndiName = getProperty(prefix + "jndiName");
353                                try {
354                                        InitialContext ctx = new InitialContext();
355                                        Context envContext  = (Context) ctx.lookup("java:/comp/env");
356                                        ds = (DataSource) envContext.lookup(jndiName);
357                                        // could fallback to global datasource if comp/env is not here
358                                } catch (NamingException e) {
359                                        throw new IllegalStateException("Could not retrieve datasource '" + jndiName + "' from JNDI", e);
360                                }
361                } else {
362                        throw new IllegalStateException("Unknown " + prefix + "connectionType property '" + connectionType + "')");
363                }
364                
365                jt = new JdbcTemplate(ds);
366                dataSources.put(connectionName, ds);
367                jdbcTemplates.put(connectionName, jt);
368        }
369        return jt;
370    }

375         * @return the security context for this application

385        /** Determines whether a user 'may have' a permission on the application. 
386         * See the hasPermission method in SecurityContext for more details.
387         *
388         * @param user The user we are checking permissions for. 
389         * @param permission The permission we are checking.
390         * @return 'true' if the user may be authorised to perform the permission,
391         *   false otherwise
392         */
393        public boolean hasPermission(User user, String permission)
394        {
395                if ("false".equals(getProperty("auth.enableSecurityContext"))) { return true; }
396                SecurityContext securityContext = getSecurityContext();
397                return securityContext.hasPermission(user, permission);
398        }
399
400        /** Determines whether a user 'may have' a permission on the application. 
401         * See the hasPermission method in SecurityContext for more details. If the user
402         * does not have the permission, then a SecurityException is thrown
403         *
404         * @see SecurityContext#hasPermission(com.randomnoun.common.security.User, String)
405         *
406         * @param user The user we are checking permissions for. 
407         * @param permission The permission we are checking.
408         * 
409         * @throws SecurityException if the user does not have the supplied permission 
410         */
411        public void checkPermission(User user, String permission) throws SecurityException {
412                if (!hasPermission(user,  permission)) {
413                        throw new SecurityException("User does not have '" + permission + "' permission");
414                }
415    }

417        /** Determines whether a user has a permission on the application. 
418         * See the hasPermission method in SecurityContext for more details.
419         *
420         * @param user The user we are checking permissions for.
421         * @param permission The permission we are checking.
422         * @param resourceContext Fine-grained resource context
423         * @return 'true' if the user is authorised to perform the permission,
424         *   false otherwise
425         */
426        public boolean hasPermission(User user, String permission, Map<String, Object> resourceContext)
427        {
428                if ("false".equals(getProperty("auth.enableSecurityContext"))) { return true; }
429                SecurityContext securityContext = getSecurityContext();
430                return securityContext.hasPermission(user, permission, resourceContext);
431        }
