168        /** Tests a resource name against the search criteria specified in this object
169         * 
170         * @param resourceName the last component of a resource name
171         * 
172         * @return true if the resource passes the search criteria, false otherwise
173         */
174        public boolean matches(String resourceName) {
175                if (resourceName==null) { throw new NullPointerException("null string"); }
176                if (ignoreCase) {
177                        switch(matchType) {
178                                case MATCHTYPE_EXACT: return searchTerm.equalsIgnoreCase(resourceName);
179                                case MATCHTYPE_REGEX: return searchPattern.matcher(resourceName).find();
180                                case MATCHTYPE_STARTSWITH: return resourceName.toUpperCase().startsWith(searchTerm);
181                                case MATCHTYPE_CONTAINS: return resourceName.toUpperCase().contains(searchTerm);
182                                default: throw new IllegalStateException("Illegal matchType '" + matchType + "'");
183                        }
184                } else {
185                        switch(matchType) {
186                                case MATCHTYPE_EXACT: return searchTerm.equals(resourceName);
187                                case MATCHTYPE_REGEX: return searchPattern.matcher(resourceName).find();
188                                case MATCHTYPE_STARTSWITH: return resourceName.startsWith(searchTerm);
189                                case MATCHTYPE_CONTAINS: return resourceName.contains(searchTerm);
190                                default: throw new IllegalStateException("Illegal matchType '" + matchType + "'");
191                        }
192                }
193        }

196        /** An {@link java.io.InputStream} wrapper which updates an internal md5/sha1 digest
197         * as the stream is being read.
198         */
199        public static class HashGeneratingInputStream extends InputStream {
200                InputStream is;
201                MessageDigest algorithm1, algorithm2;
202                
203                public HashGeneratingInputStream(InputStream is) {
204                        this.is = is;
205                        try {
206                                algorithm1 = MessageDigest.getInstance("MD5");
207                                algorithm2 = MessageDigest.getInstance("SHA-1");
208                        } catch (NoSuchAlgorithmException nsae) {
209                                throw (IllegalStateException) new IllegalStateException(
210                                        "Invalid crypto config").initCause(nsae);
211                        }
212                algorithm1.reset();
213                algorithm2.reset();
214
215                }
216                
217                @Override
218                public int read() throws IOException {
219                        int result = is.read();
220                        if (result != -1) {
221                                algorithm1.update((byte) result);
222                                algorithm2.update((byte) result);
223                        }
224                        return result;
225                }
226                public int available() throws IOException {
227                        return is.available();
228                }
229                public void close() {
230                        // ignored;
231                }
232                public void mark(int readlimit) {
233                        is.mark(readlimit);
234                }
235        public boolean markSupported() {
236                return is.markSupported();
237        }
238        public int read(byte[] b)  throws IOException {
239                int result = is.read(b);
240                if (result!=-1) {
241                        algorithm1.update(b, 0, result);
242                        algorithm2.update(b, 0, result);
243                }
244                return result;
245        }
246        public int read(byte[] b, int off, int len)  throws IOException {
247                int result = is.read(b, off, len);
248                if (result!=-1) {
249                        algorithm1.update(b, off, result);
250                        algorithm2.update(b, off, result);
251                }
252                        return result;
253        }
254        public void reset() throws IOException {
255                is.reset();
256        }
257        public long skip(long n) throws IOException {
258                return is.skip(n);
259        }
260        /** Returns the MD5 digest of all input that has been read by this InputStream so far,
261         * in a hexadecimal String form */
262        public String getMd5() {
263                byte messageDigest[] = algorithm1.digest();
264                //System.err.println("md5 messageDigest is " + messageDigest.length + " bytes");
265                StringBuffer hexString = new StringBuffer();
266                for (int i=0;i<messageDigest.length;i++) {
267                        hexString.append(Integer.toString( ( messageDigest[i] & 0xff ) + 0x100, 16).substring( 1 ));
268                }
269                return hexString.toString();
270        }
271        /** Returns the SHA1 digest of all input that has been read by this InputStream so far,
272         * in a hexadecimal String form */
273                public String getSha1() {
274                byte messageDigest[] = algorithm2.digest();
275                //System.err.println("sha1 messageDigest is " + messageDigest.length + " bytes");
276                StringBuffer hexString = new StringBuffer();
277                for (int i=0;i<messageDigest.length;i++) {
278                        hexString.append(Integer.toString( ( messageDigest[i] & 0xff ) + 0x100, 16).substring( 1 ));
279                }
280                return hexString.toString();
281        }
282        }

334        /** Class which defines a callback which sends names and resource hashes to System.out.
335         */
336        public static class HashingResourceFinderCallback implements ResourceFinderCallback {
337                
338                int resourceIndex = 0;
339                boolean showHashes = false;
340        
341                public HashingResourceFinderCallback() {
342                }
343                
344                public ResourceFinderCallbackResult preProcess(String resourceName, long filesize, long timestamp, InputStream inputStream) throws IOException {
345                        ResourceFinderCallbackResult rfcbr = new ResourceFinderCallbackResult();
346                        rfcbr.replaceInputStream = new HashGeneratingInputStream(inputStream);
347                        return rfcbr;
348                }
349                
350                public ResourceFinderCallbackResult postProcess(String resourceName, long filesize, long timestamp, InputStream inputStream, ResourceFinderCallbackResult preProcessResult) throws IOException {
351                        ResourceFinderCallbackResult rfcbr = new ResourceFinderCallbackResult();
352                        try {
353                                SimpleDateFormat sdf = new SimpleDateFormat("dd/MM/yyyy HH:mm:ss");
354                                String md5, sha1;
355                                if (inputStream instanceof HashGeneratingInputStream) {
356                                        HashGeneratingInputStream hgis = (HashGeneratingInputStream) inputStream;
357                                        // pump the rest of the bits through this stream
358                                        byte[] buffer = new byte[4096];
359                                        while (inputStream.read(buffer) != -1) { /* nothing */ }
360                                        md5 = hgis.getMd5();
361                                        sha1 = hgis.getSha1();
362                                } else {
363                                        MessageDigest algorithm1, algorithm2;
364                                        try {
365                                                algorithm1 = MessageDigest.getInstance("MD5");
366                                                algorithm2 = MessageDigest.getInstance("SHA1");
367                                        } catch (NoSuchAlgorithmException nsae) {
368                                                throw (IllegalStateException) new IllegalStateException(
369                                                        "Invalid crypto config").initCause(nsae);
370                                        }
371                                algorithm1.reset();
372                                algorithm2.reset();
373                                        byte[] buffer = new byte[4096];
374                                        int bytesRead;
375                                        while ((bytesRead = inputStream.read(buffer)) != -1) {
376                                                algorithm1.update(buffer, 0, bytesRead);
377                                                algorithm2.update(buffer, 0, bytesRead);
378                                }
379                                byte messageDigest1[] = algorithm1.digest();
380                                byte messageDigest2[] = algorithm2.digest();
381                                StringBuffer hexString1 = new StringBuffer();
382                                StringBuffer hexString2 = new StringBuffer();
383                                for (int i=0; i<messageDigest1.length; i++) {
384                                        hexString1.append(Integer.toString( ( messageDigest1[i] & 0xff ) + 0x100, 16).substring( 1 ));
385                                }
386                                for (int i=0; i<messageDigest2.length; i++) {
387                                        hexString2.append(Integer.toString( ( messageDigest2[i] & 0xff ) + 0x100, 16).substring( 1 ));
388                                }
389                                md5 = hexString1.toString();
390                                sha1 = hexString2.toString();
391                                }
392                        System.out.println("[" + resourceIndex + "] " + resourceName + " " + (filesize==-1 ? "(unknown)" : String.valueOf(filesize)) + 
393                                " " + sdf.format(new Date(timestamp)) + " " + md5 + " " + sha1 );
394                        } catch (IllegalArgumentException iae) {
395                                // not sure if this is needed any more
396                                throw new IOException("IllegalArgumentException processing ZipInputStream", iae);
397                        }
398                        resourceIndex++;
399                        return rfcbr;
400                }
401        }

404        /** Class which defines a callback which sends names and resources to System.out.
405         * 
406         * <p>This class uses '#' as a separator between the filesystem and files contained
407         * within archives; e.g. test.jar#abc.txt refers to abc.txt in test.jar.
408         * 
409         * <p>For comparison, Tangosol seems to use '!', includes a leading slash and 
410         * includes a protocol-like identifier at the beginning
411         * (e.g. jar:file:test.jar!/abc.txt). If a constructor is supplied which only provides
412         * a ZipInputStream (i.e. no filename is available), then resources will be returned 
413         * starting with the '#' character.
414         * 
415         */
416        public static class DisplayResourceFinderCallback implements ResourceFinderCallback {
417                
418                public final static int DUMP_NAMES = -1;
419                public final static int DUMP_NAMES_AND_RESOURCES = -2;
420                public final static int DUMP_RESOURCES = 0;
421        
422                int dumpType = 0;
423                List<Integer> dumpResourceNumbers;
424                int maxDumpResourceNumber = -1;
425                int resourceIndex = 0;
426                boolean verbose = false;
427                boolean manifests = false;
428                boolean decompile = false;
429                String searchContents = null;
430                boolean searchContentsIgnoreCase = false;
431
432                // with all the trimmings
433                public DisplayResourceFinderCallback(int dumpType, List<Integer> dumpResourceNumbers, boolean verbose, boolean manifests, boolean decompile, String searchContents, boolean searchContentsIgnoreCase) {
434                        // System.out.println("2 searchContents is " + searchContents);
435                        this.dumpResourceNumbers = dumpResourceNumbers;
436                        this.dumpType = dumpType;
437                        this.verbose = verbose;
438                        this.manifests = manifests;
439                        this.decompile = decompile;
440                        this.searchContents = searchContents;
441                        this.searchContentsIgnoreCase = searchContentsIgnoreCase;
442                        if (dumpResourceNumbers != null) {
443                                for (Integer drn : dumpResourceNumbers) {
444                                        maxDumpResourceNumber = Math.max(maxDumpResourceNumber, drn.intValue());
445                                }
446                        }
447                }
448                
449                private ResourceFinderCallbackResult dump(String resourceName, long filesize, long timestamp, InputStream inputStream) throws IOException {
450                        ResourceFinderCallbackResult rfcr = new ResourceFinderCallbackResult(); 
451                        if ((dumpType == DUMP_NAMES_AND_RESOURCES || dumpType == DUMP_NAMES) &&
452                                (dumpResourceNumbers==null || dumpResourceNumbers.contains(new Integer(resourceIndex)))
453                           )
454                        {
455                                if (verbose) {
456                                        SimpleDateFormat sdf = new SimpleDateFormat("dd/MM/yyyy HH:mm:ss");
457                                        System.out.println("[" + resourceIndex + "] " + resourceName + " " + (filesize==-1 ? "(unknown)" : String.valueOf(filesize)) + " " + sdf.format(new Date(timestamp)));
458                                } else if (searchContents == null) {
459                                        System.out.println("[" + resourceIndex + "] " + resourceName);
460                                } else if (searchContents != null) {
461                                        int pos = -1;
462                                        // @TODO this assumes we never search for strings containing newlines
463                                        LineNumberReader lnr = new LineNumberReader(new InputStreamReader(inputStream));
464                                        if (searchContentsIgnoreCase) {
465                                                searchContents = searchContents.toLowerCase();
466                                                String line = lnr.readLine();
467                                                if (line!=null) {
468                                                        pos = line.toString().toLowerCase().indexOf(searchContents.toLowerCase());
469                                                        while (line!=null && pos==-1) {
470                                                                line = lnr.readLine();
471                                                                pos = line==null ? -1 : line.toString().toLowerCase().indexOf(searchContents.toLowerCase());
472                                                        }
473                                                }
474                                        } else {
475                                                String line = lnr.readLine();
476                                                if (line!=null) {
477                                                        pos = line.toString().indexOf(searchContents);
478                                                        while (line!=null && pos==-1) {
479                                                                line = lnr.readLine();
480                                                                pos = line==null ? -1 :  line.toString().indexOf(searchContents);
481                                                        }
482                                                }
483                                        }
484                                        if (pos!=-1) {
485                                                System.out.println("[" + resourceIndex + "] [line " + lnr.getLineNumber() + ", col " + pos + "] " + resourceName);
486                                        }
487                                }
488                        }  
489                        if ((dumpType == DUMP_NAMES_AND_RESOURCES || dumpType == DUMP_RESOURCES) &&
490                                (dumpResourceNumbers==null || dumpResourceNumbers.contains(new Integer(resourceIndex)))
491                                ) {
492                                // InputStream is = ResourceFinder.getResourceStream(resourceName);
493                                if (manifests) {
494                                        BufferedReader br = new BufferedReader(new InputStreamReader(inputStream));
495                                        String line = br.readLine();
496                                        while (line != null) {
497                                                int len = line.length();
498                                                while (line.length() > 0 && line.charAt(0) == ' ') {
499                                                        line = line.substring(1);
500                                                }
501                                                System.out.print(line);
502                                                if (len != 70) { System.out.println(); }
503                                                line = br.readLine();
504                                        }
505                                } else if (decompile) {
506                                        // hopefully this is deleted when the VM exits
507                                        // @TODO: we need to grab all inner classes for this class as well
508                                        File tmpFile = File.createTempFile("resourceFinder", ".class");
509                                        FileOutputStream fos = new FileOutputStream(tmpFile); 
510                                        StreamUtil.copyStream(inputStream, fos, 1024);
511                                        fos.close();
512                                        try {
513                                                String result = ProcessUtil.exec(new String[] { "jad", "-lnc", "-p", tmpFile.getCanonicalPath() });
514                                                System.out.println(result);
515                                        } catch (ProcessUtil.ProcessException pe) {
516                                                throw (IOException) new IOException("Problem executing jad").initCause(pe);
517                                        }
518                                        
519                                } else {
520                                        StreamUtil.copyStream(inputStream, System.out, 1024);
521                                }
522                                if (resourceIndex == maxDumpResourceNumber) {
523                                        // don't bother continuing this search if we're found the last resource being searched for
524                                        rfcr.abort = true;
525                                }
526
527                                // we don't insert an additional newline when dumping the contents of just one file
528                                // so that stdout redirection still does something useful
529                                if (dumpType == DUMP_NAMES_AND_RESOURCES && 
530                                        (dumpResourceNumbers==null || dumpResourceNumbers.size()>1)) {
531                                        System.out.println();
532                                }
533                        }
534                        resourceIndex++;
535                        return rfcr;
536                }

567        /** Common code to both the File and ZipInputStream constructors */
568        private void init(String searchTerm, int matchType, boolean ignoreCase, 
569                ResourceFinderCallback callback) 
570        {
571                this.matchType = matchType;
572                this.ignoreCase = ignoreCase;
573                this.showAll = false;
574                this.ignoreErrors = false;
575                this.callback = callback;
576
577                // @TODO clean this up a bit
578                if (ignoreCase) {
579                        switch (matchType) {
580                                case MATCHTYPE_EXACT: break;
581                                case MATCHTYPE_REGEX: searchPattern = Pattern.compile(searchTerm, Pattern.CASE_INSENSITIVE); break;
582                                case MATCHTYPE_STARTSWITH: searchTerm = searchTerm.toUpperCase(); break;
583                                case MATCHTYPE_CONTAINS: searchTerm = searchTerm.toUpperCase(); break;
584                                default: throw new IllegalStateException("Illegal matchType '" + matchType + "'");
585                        }
586                } else {
587                        switch (matchType) {
588                                case MATCHTYPE_EXACT: break;
589                                case MATCHTYPE_REGEX: searchPattern = Pattern.compile(searchTerm); break;
590                                case MATCHTYPE_STARTSWITH: break;
591                                case MATCHTYPE_CONTAINS: break;
592                                default: throw new IllegalStateException("Illegal matchType '" + matchType + "'");
593                        }
594                }
595                this.searchTerm = searchTerm;
596                
597        }

848        /** Returns a list of resources within the supplied archive, 
849         * and archives contained within this archive
850         * 
851         * @param zipInputStream the archive to search
852         * @param prefix a prefix which is included in any results returned by this
853         *   method. By convention, this prefix should end with a '#' to separate it
854         *   from resources found within the resource.
855         * 
856         * @throws IOException 
857         */
858
859        public void findResourceInZip(ZipInputStream zipInputStream, String prefix) throws IOException {
860                if (maxArchiveDepth!=-1 && currentArchiveDepth>=maxArchiveDepth) {
861                        return;
862                }
863                // System.err.println("findResourceInZip(" + prefix + "):" + currentArchiveDepth);
864                currentArchiveDepth++;
865                
866                // System.out.println("Searching in " + prefix);
867                ZipEntry zipEntry = null;
868                try {
869                        zipEntry = zipInputStream.getNextEntry();
870                } catch (EOFException oefe) {
871                        ignorableException("Error retrieving first entry in zip '" + prefix.substring(0, prefix.length()-1) + "'", oefe);
872                        currentArchiveDepth--;
873                        return;
874                } catch (ZipException ze) {
875                        ignorableException("Error retrieving first entry in zip '" + prefix.substring(0, prefix.length()-1) + "'", ze);
876                        currentArchiveDepth--;
877                        return;
878                } catch (IllegalArgumentException iae) {
879                        // can occur in ZipInputStream.getUTF8String
880                        ignorableException("Error retrieving first entry in zip '" + prefix.substring(0, prefix.length()-1) + "'", iae);
881                        currentArchiveDepth--;
882                        return;
883                }
884                while (zipEntry != null) {
885                        String name = zipEntry.getName();
886                        String shortName = name;
887                        // zipEntry.isDirectory(); // write these at the end ? skip them altogether ?
888
889                        // on unix, it's possible to get directory entries (trailing '/'s) within ZIPs; on windows this doesn't seem to happen
890                        while (shortName.endsWith("/")) { shortName = shortName.substring(0, shortName.length() - 1); }
891                        while (shortName.endsWith("\\")) { shortName = shortName.substring(0, shortName.length() - 1); }
892                        if (shortName.indexOf('/')!=-1) { shortName = shortName.substring(shortName.lastIndexOf('/') + 1); }
893                        if (shortName.indexOf('\\')!=-1) { shortName = shortName.substring(shortName.lastIndexOf('\\') + 1); }
894
895                        // may need to do case-sensitive match
896                        /* commenting this out temporarily
897
898                        if (showVersions && name.equalsIgnoreCase("META-INF/MANIFEST.MF")) {
899                                // treat this as a property file. Which is wrong, because it's got insane line breaks
900                                // but good enough for retrieving version data
901                                
902                                Properties props = new Properties();
903                                props.load(zipInputStream);
904                                if (props.getProperty("Specification-Version")!=null) {
905                                        // there's also an Implementation-Version, but this appears to be the same
906                                        // maven2 doesn't write these entries. perhaps.
907                                        // @TODO something
908                                }
909                        }
910                        */
911                        
912                        InputStream inputStreamToProcess = zipInputStream;
913                        
914                        // might just be easier to add .reset() to ZipInputStream
915                        ResourceFinderCallbackResult rfcbResult = null;
916                        if (isArchive(name)) {
917                                try {
918                                        if (matches(shortName) || showAll) {
919                                                rfcbResult = callback.preProcess(prefix + name, zipEntry.getSize(), zipEntry.getTime(), inputStreamToProcess);
920                                                if (rfcbResult!=null && rfcbResult.replaceInputStream!=null) {
921                                                        inputStreamToProcess = rfcbResult.replaceInputStream;
922                                                }
923                                                if (rfcbResult!=null && rfcbResult.abort) {
924                                                        this.abort = true; break;
925                                                }
926                                        }
927                                        ZipInputStream zis = new ZipInputStream(inputStreamToProcess);
928                                        findResourceInZip(zis, prefix + name + "#");
929                                        zipInputStream.closeEntry();
930                                } catch (EOFException eofe) {
931                                        // can trigger "java.io.EOFException: Unexpected end of ZLIB input stream" errors
932                                        ignorableException("Error reading zip '" + prefix + name + "'", eofe);
933                                } catch (ZipException ze) {
934                                        ignorableException("Error reading zip '" + prefix + name + "'", ze);
935                                }
936                        }
937                        if (matches(shortName) || showAll) {
938                                rfcbResult = callback.postProcess(prefix + name, zipEntry.getSize(), zipEntry.getTime(), inputStreamToProcess, rfcbResult);
939                                if (rfcbResult!=null && rfcbResult.abort) {
940                                        this.abort = true; break;
941                                }
942                        }
943                        
944                        try {
945                                zipEntry = zipInputStream.getNextEntry();
946                        } catch (EOFException oefe) {
947                                ignorableException("Error retrieving next entry in zip '" + prefix.substring(0, prefix.length()-1) + "'; after '"+ name + "'", oefe);
948                                break;
949                        } catch (ZipException ze) {
950                                ignorableException("Error retrieving next entry in zip '" + prefix.substring(0, prefix.length()-1) + "'; after '"+ name + "'", ze);
951                                break;
952                        } catch (IllegalArgumentException iae) {
953                                // can occur in ZipInputStream.getUTF8String
954                                ignorableException("Error retrieving next entry in zip '" + prefix.substring(0, prefix.length()-1) + "'; after '"+ name + "'", iae);
955                                break;
956                        } catch (IOException ioe) {
957                                // may occur after dodgy CRCs:
958                                // invalid entry CRC (expected 0xab633fa2 but got 0xc30a2df7)
959                                // Exception in thread "main" java.io.IOException: Push back buffer is full
960                                if (ioe.getMessage().contains("Push back buffer")) {
961                                        ignorableException("Error retrieving next entry in zip '" + prefix.substring(0, prefix.length()-1) + "'; after '"+ name + "'", ioe);
962                                        break;
963                                } else {
964                                        currentArchiveDepth--;
965                                        throw ioe;
966                                }
967                        }
968                        if (abort) { break ; }
969                }
970                currentArchiveDepth--;
971        }

974         * 
975         * @param resourceName a resource name, as defined by the class javadoc
976         *

978         * 
979         * @throws FileNotFoundException the resource could not be found
980         * @throws IOException the resource could not be read

982        public static InputStream getResourceStream(String resourceName) throws IOException {
983                int pos = resourceName.indexOf("#");
984                if (pos == -1) {
985                        return new FileInputStream(resourceName);
986                } else {
987                        String filename = resourceName.substring(0, pos);
988                        String component = resourceName.substring(pos + 1);
989                        ZipInputStream zis = new ZipInputStream(new FileInputStream(filename));
990                        return getResourceComponent(zis, component, resourceName);
991                }
992        }

994        /** Private method to recursively search within an archive for a file
995         * 
996         * @param zipInputStream input stream to search
997         * @param component resource name fragment, separated by '#' characters
998         * @param fullResource full resource name (only used in exception messages)  
999         * 
1000         * @return the input stream 
1001         * 
1002         * @throws IOException the input stream cannot be read
1003         */
1004        public static InputStream getResourceComponent(ZipInputStream zipInputStream, String component, String fullResource) throws IOException {
1005                int pos = component.indexOf("#");
1006                String filename = component;
1007                String subComponent = null;
1008                if (pos != -1) {
1009                        filename = component.substring(0, pos);
1010                        subComponent = component.substring(pos + 1);
1011                }
1012                ZipEntry zipEntry = zipInputStream.getNextEntry();
1013                while (zipEntry != null) {
1014                        if (zipEntry.getName().equals(filename)) {
1015                                if (subComponent == null) {
1016                                        return zipInputStream;
1017                                } else {
1018                                        return getResourceComponent(new ZipInputStream(zipInputStream), subComponent, fullResource); 
1019                                }
1020                        }
1021                        zipEntry = zipInputStream.getNextEntry();
1022                }
1023                throw new FileNotFoundException("Could not find component '" + filename + "' in '" + fullResource + "'");
1024        }

1221                }

1232                resourceFinder.setFollowSymLinks(followSymlinks);

1234                resourceFinder.setIgnoreErrors(ignoreErrors);
