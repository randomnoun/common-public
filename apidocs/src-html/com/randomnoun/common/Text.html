074    /** Returns true if the supplied string is non-null and only contains numeric characters
075    * or a single decimal point. The value can have a leading negative ('-') symbol.
076    * 
077    * @param text The string to test
078    * @return true if the supplied string is non-null and only contains numeric characters,
079    *   which may contain a '.' character in there somewhere.

081   public static boolean isNumericDecimal(String text) {
082       if (text == null) {
083           return false;
084       }
085       boolean seenPoint = false; // existential quandary there for you
086       char ch;
087       int len = text.length();
088       for (int i = 0; i < len; i++) {
089           ch = text.charAt(i);
090           if (ch=='.') {
091                   if (seenPoint) { return false; }
092                   seenPoint = true;
093           } else if (ch == '-' && i == 0) {
094                   // leading negative symbol OK
095                   if (len == 1) {
096                           // but not if it's the only character in the string
097                           return false;
098                   }
099           } else if (ch < '0' || ch > '9') {
100               return false;
101           }
102       }
103       return true;
104   }

118        }
119            boolean seenPoint = false; // existential quandary there for you

121            char ch;
122            for (int i = 0; i < text.length(); i++) {
123                ch = text.charAt(i);
124                if (ch=='E') {
125                        if (expPos != -1) { return false; }

133                // leading negative symbol OK
134            } else if (ch < '0' || ch > '9') {
135                return false;
136            }
137            }
138            return true;

142    /** Ensures that a string returned from a browser (on any platform) conforms
143     * to unix line-EOF conventions. Any instances of consecutive CRs (<tt>0xD</tt>) 
144     * and LFs (<tt>0xA</tt>) in a string will be reduced to a series of CRs (the number of CRs will be the
145     * maximum number of CRs or LFs found in a row).  
146     * 
147     * @param input the input string
148     * 
149     * @return the canonicalised string, as described above
150     */
151    public static String reduceNewlines(String input) {
152        StringBuilder sb = new StringBuilder();
153        int len = input.length();
154        int crCount = 0;
155                int lfCount = 0;
156                boolean insertNewline = false;
157                char ch;
158        for (int i=0; i<len; i++) {
159                ch = input.charAt(i);
160                if (ch == (char) 0xA) {
161                        lfCount ++; insertNewline = true;
162                } else if (ch == (char) 0xD) {
163                        crCount ++; insertNewline = true;
164                } else if (insertNewline) {
165                                for (int j=0; j<Math.max(lfCount, crCount); j++) {
166                                        sb.append((char) 0xA);
167                                }
168                                insertNewline = false; lfCount=0; crCount=0;
169                                sb.append(ch);
170                } else {
171                        sb.append(ch);
172                }
173        }
174        if (insertNewline) {
175                        for (int j=0; j<Math.max(lfCount, crCount); j++) {
176                                sb.append((char) 0xA);
177                        }
178        }
179        
180        return sb.toString();
181    }

185     * Returns the HTML-escaped form of a string. The <tt>&amp;</tt>,
186     * <tt>&lt;</tt>, <tt>&gt;</tt>, and <tt>"</tt> characters are converted to
187     * <tt>&amp;amp;</tt>, <tt>&amp;lt;<tt>, <tt>&amp;gt;<tt>, and
188     * <tt>&amp;quot;</tt> respectively.
189     * 
190     * <p>Characters in the unicode control code blocks ( apart from \t, \n and \r ) are converted to &xfffd;
191     * <p>Characters outside of the ASCII printable range are converted into &xnnnn; form
192     *
193     * @param string the string to convert
194     *
195     * @return the HTML-escaped form of the string
196     */
197    static public String escapeHtml(String string) {
198        if (string == null) {
199            return "";
200        }
201        char c;
202        String hex;
203        StringBuilder sb = new StringBuilder(string.length());
204        for (int i = 0; i < string.length(); i++) {
205            c = string.charAt(i);
206            // check for illegal characters
207            switch (c) {
208                case '&':
209                    sb.append("&amp;");
210                    break;
211                case '<':
212                    sb.append("&lt;");
213                    break;
214                case '>':
215                    sb.append("&gt;");
216                    break;
217                case '\"':
218                    // interestingly, &quote; (with the e) works fine for HTML display,
219                    // but not inside hidden field values
220                    sb.append("&quot;");
221                    break;
222                default:
223                        // 'illegal characters' according to ESAPI. 7f to 9f are control characters in unicode 
224                        if ( ( c <= 0x1f && c != '\t' && c != '\n' && c != '\r' ) || ( c >= 0x7f && c <= 0x9f ) ) {
225                                sb.append("&#xfffd;"); // REPLACEMENT_HEX in ESAPI's HtmlEntityCodec
226                        } else if ( c > 0x1f && c <= 0x7f ) {
227                                // safe printable
228                                sb.append(c);
229                        } else {
230                                // ESAPI didn't have the else block above, which was causing it escape everything 
231                                hex = getHexForNonAlphanumeric(c);
232                                sb.append("&#x" + hex + ";");
233                        }
234                        
235            }
236        }
237
238        return sb.toString();
239    }

305    /**
306     * Returns the csv-escaped form of a string. A csv-escaped string is
307     * used when writing to a CSV (comma-separated-value) file. It ensures
308     * that commas included within a string are quoted. We use the Microsoft-Excel
309     * quoting rules, so that our CSV files can be imported into that. These rules
310     * (derived from experimentation) are:
311     *
312     * <ul>
313     * <li>Strings without commas (,) inverted commas ("), or newlines (\n) are returned as-is.
314     * <li>Otherwise, the string is surrounded by inverted commas, and any
315     *   inverted commas within the string are doubled-up (i.e. '"' becomes '""').
316     * <li>A value that starts with any of "=", "@", "+" or "-" has a leading single apostrophe added
317     *   to prevent the value being evaluated in Excel. The leading quote is visible to the user when the
318     *   csv is opened, which may mean that it will have to be removed when roundtripping data.
319     *   This may complicate things if the user actually wants a leading single quote in their CSV value.   
320     * </ul>
321     *
322     * <p>Embedded newlines are inserted as-is, as per Excel. This will require
323     * some care whilst parsing if we want to be able to read these files.
324     *
325     * @param string the string to convert
326     *
327     * @return the csv-escaped form of the string
328     */
329    static public String escapeCsv(String string) {
330        if (string == null) {
331            return "";
332        }
333
334        boolean quoted = false;
335        // from https://www.contextis.com/en/blog/comma-separated-vulnerabilities
336        // prefix cells that start with ‘=’ , '@', '+' or '-' with an apostrophe 
337        // This will ensure that the cell isn’t interpreted as a formula, and as a bonus in Microsoft Excel the apostrophe itself will not be displayed.
338        if (string.startsWith("=") || 
339          string.startsWith("@")) {
340            // prefix the string with an a single quote char to escape it
341            string = "'" + string;
342            quoted = true; // not sure need to quote here, but doesn't hurt
343        } else if ((string.startsWith("+") || string.startsWith("-")) && 
344                (string.length() == 1 || !Text.isNumericDecimalExp(string))) {
345                // numbers can legitimately start with '+' or '-' but anything else should be escaped
346                string = "'" + string;
347            quoted = true; 
348        }
349
350        
351        if (string.indexOf(',') == -1 && string.indexOf('"') == -1 && string.indexOf('\n') == -1 && !quoted) {
352                return string;
353        }
354        string = Text.replaceString(string, "\"", "\"\"");
355        string = "\"" + string + "\"";
356
357        return string;
358    }

360    /** Given a csv-encoded string (as produced by the rules in {@see #escapeCsv},
361     *  produces a List of Strings which represent the individual values in the string.
362     *  Note that this method is *not* equivalent to calling <tt>Arrays.asList(astring.split(","))</tt>.
363     *
364     * <p>Setting the whitespaceSensitive parameter to false allows leading and trailing
365     * whitespace in *non-quoted* values to be removed, e.g. if the input string <tt>text</tt> is:
366     *
367     * <pre class="code">
368     * abc,def,  ghi, j k ,"lmn"," op "," q,r","""hello""", "another"
369     * </pre>
370     *
371     * then <tt>parseCsv(text, <b>false</b>)</tt> will return the strings:
372     * <pre class="code">
373     * abc
374     * def
375     * ghi
376     * j k
377     * lmn
378     *  op        <i>(this String has one leading space, and a trailing space after 'p')</i>
379     *  q,r       <i>(this String has one leading space)</i>
380     * "hello"
381     * another
382     * </pre>
383     *
384     * and <tt>parseCsv(text, <b>true</b>)</tt> would throw a ParseException (since the
385     * final element is a quoted value, but begins with a space).
386     *
387     * If the <tt>, "another"</tt> text is removed, however, then
388     * <tt>parseCsv(text, true)</tt> would return the following:
389     *
390     * and <tt>parseCsv(text, true)</tt> will return the string
391     * <pre>
392     * abc
393     * def
394     *   ghi      <i>(this String has two leading spaces)</i>
395     *  j k       <i>(this String has one leading space and a trailing space after the 'k' character)</i>
396     * lmn
397     *  op        <i>(this String has one leading space, and a trailing space after 'p')</i>
398     *  q,r       <i>(this String has one leading space)</i>
399     * "hello"
400     * </pre>
401     *
402     * <p>Most applications would want to use the 'whiteSpaceSensitive=false' form of this function, since
403     * (a) less chance of a ParseException, and (b) it's what an end-user would normally
404     * expect. This can be performed by calling the {@link #parseCsv(String)} method.
405     *
406     * <p>Whitespace is determined by using the <tt>Character.isSpaceChar()</tt> method,
407     * which is Unicode-aware.
408     *
409     * @param text   The CSV-encoded string to parse
410     * @param trim   If set to true, will trim leading and trailing whitespace in *non-quoted* values.
411     *
412     * @return a List of Strings. The returned List is guaranteed to always contain at least one element.
413     *
414     * @throws NullPointerException if the text passed to this method is null
415     * @throws ParseException if a quoted value contains leading whitespace before the
416     *  opening quote, or after the trailing quote.
417     * @throws ParseException if a quoted value has a start quote, but no end quote, or
418     *   if a value has additional text after a quoted value (before the next comma or EOL).
419     */
420    static public List<String> parseCsv(String text, boolean whitespaceSensitive)
421        throws ParseException {
422        if (text == null) {
423            throw new NullPointerException("null text");
424        }
425
426        // parse state: 
427        //   0=searching for new value (at start of line or after comma) 
428        //   1=consuming non-quoted values
429        //   2=consuming quoted value
430        //   3=consumed first quote within a quoted value (may be termining quote or a "" sequence)
431        //   4=consuming whitespace up to next comma/EOL (after quoted value, not whitespaceSensitive)
432        int parseState = 0;
433        int length = text.length();
434        String element;
435        List<String> result = new ArrayList<String>();
436        char ch;
437        StringBuilder buffer = new StringBuilder();
438
439        for (int pos = 0; pos < length; pos++) {
440            ch = text.charAt(pos);
441
442            // System.out.println("pos " + pos + ", state=" + parseState + ", nextchar=" + ch + ", buf=" + buffer);
443            switch (parseState) {
444                case 0:
445                    if (Character.isSpaceChar(ch)) {
446                        if (whitespaceSensitive) {
447                            buffer.append(ch);
448                            parseState = 1;
449                        } else {
450                            // ignore
451                        }
452                    } else if (ch == '"') {
453                        parseState = 2;
454                    } else if (ch == ',') {
455                        result.add(""); // add an empty element; state remains unchanged
456                    } else {
457                        buffer.append(ch);
458                        parseState = 1;
459                    }
460                    break;
461                case 1:
462                    if (ch == ',') {
463                        element = buffer.toString();
464                        if (!whitespaceSensitive) {
465                            element = element.trim();
466                        }
467                        result.add(element);
468                        buffer.setLength(0);
469                        parseState = 0;
470                    } else {
471                        buffer.append(ch);
472                    }
473                    break;
474                case 2:
475                    if (ch == '"') {
476                        parseState = 3;
477                    } else {
478                        buffer.append(ch);
479                    }
480                    break;
481                case 3:
482                    if (ch == '"') {
483                        buffer.append('"');
484                        parseState = 2;
485                    } else if (ch == ',') {
486                        result.add(buffer.toString());
487                        buffer.setLength(0);
488                        parseState = 0;
489                    } else if (Character.isSpaceChar(ch)) {
490                        if (whitespaceSensitive) {
491                            throw new ParseException("Cannot have trailing whitespace after close quote character", pos);
492                        }
493                        parseState = 4;
494                    } else {
495                        throw new ParseException("Cannot have trailing data after close quote character", pos);
496                    }
497                    break;
498                case 4:
499                    if (Character.isSpaceChar(ch)) {
500                        // consume and ignore
501                    } else if (ch == ',') {
502                        result.add(buffer.toString());
503                        buffer.setLength(0);
504                        parseState = 0;
505                    } else {
506                        throw new ParseException("Cannot have trailing data after close quote character", pos);
507                    }
508                    break;
509                    
510                default:
511                    throw new IllegalStateException("Illegal state '" + parseState + "' in parseCsv");
512            }
513        }
514
515        // if state is 2, we are in the middle of a quoted value
516        if (parseState == 2) {
517            throw new ParseException("Missing endquote in csv text", length);
518        }
519
520        // otherwise we still need to add what's left in the buffer into the result list
521        element = buffer.toString();
522        if (parseState == 1 && !whitespaceSensitive) {
523            element = element.trim();
524        }
525        result.add(element);
526        return result;
527    }

1709    /** Number of character edits between two strings; taken from  
1710         * http://www.merriampark.com/ldjava.htm. There's a version in commongs-lang,
1711         * apparently, but according to the comments on that page, it uses O(n^2) memory,
1712         * which can't be good.
1713         * 
1714         * @param s string 1
1715         * @param t string 2
1716         *  
1717         * @return the smallest number of edits required to convert s into t 
1718         */
1719        public static int getLevenshteinDistance (String s, String t) {
1720                  if (s == null || t == null) {
1721                    throw new IllegalArgumentException("Strings must not be null");
1722                  }
1723                                
1724                  /*
1725                    The difference between this impl. and the previous is that, rather 
1726                     than creating and retaining a matrix of size s.length()+1 by t.length()+1, 
1727                     we maintain two single-dimensional arrays of length s.length()+1.  The first, d,
1728                     is the 'current working' distance array that maintains the newest distance cost
1729                     counts as we iterate through the characters of String s.  Each time we increment
1730                     the index of String t we are comparing, d is copied to p, the second int[].  Doing so
1731                     allows us to retain the previous cost counts as required by the algorithm (taking 
1732                     the minimum of the cost count to the left, up one, and diagonally up and to the left
1733                     of the current cost count being calculated).  (Note that the arrays aren't really 
1734                     copied anymore, just switched...this is clearly much better than cloning an array 
1735                     or doing a System.arraycopy() each time  through the outer loop.)
1736
1737                     Effectively, the difference between the two implementations is this one does not 
1738                     cause an out of memory condition when calculating the LD over two very large strings.              
1739                  */            
1740                                
1741                  int n = s.length(); // length of s
1742                  int m = t.length(); // length of t
1743                                
1744                  if (n == 0) {
1745                    return m;
1746                  } else if (m == 0) {
1747                    return n;
1748                  }
1749
1750                  int p[] = new int[n+1]; //'previous' cost array, horizontally
1751                  int d[] = new int[n+1]; // cost array, horizontally
1752                  int _d[]; //placeholder to assist in swapping p and d
1753
1754                  // indexes into strings s and t
1755                  int i; // iterates through s
1756                  int j; // iterates through t
1757
1758                  char t_j; // jth character of t
1759
1760                  int cost; // cost
1761
1762                  for (i = 0; i<=n; i++) {
1763                     p[i] = i;
1764                  }
1765                                
1766                  for (j = 1; j<=m; j++) {
1767                     t_j = t.charAt(j-1);
1768                     d[0] = j;
1769                                
1770                     for (i=1; i<=n; i++) {
1771                        cost = s.charAt(i-1)==t_j ? 0 : 1;
1772                        // minimum of cell to the left+1, to the top+1, diagonally left and up +cost                            
1773                        d[i] = Math.min(Math.min(d[i-1]+1, p[i]+1),  p[i-1]+cost);  
1774                     }
1775
1776                     // copy current distance counts to 'previous row' distance counts
1777                     _d = p;
1778                     p = d;
1779                     d = _d;
1780                  } 
1781                                
1782                  // our last action in the above loop was to switch d and p, so p now 
1783                  // actually has the most recent cost counts
1784                  return p[n];
1785        }
