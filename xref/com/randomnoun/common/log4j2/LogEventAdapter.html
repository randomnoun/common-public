17  package com.randomnoun.common.log4j2;
18  
19  import org.apache.log4j.Category;
20  import org.apache.log4j.Level;
21  import org.apache.log4j.spi.LocationInfo;
22  import org.apache.log4j.spi.LoggingEvent;
23  import org.apache.log4j.spi.ThrowableInformation;
24  import org.apache.logging.log4j.core.LogEvent;
25  import org.apache.logging.log4j.spi.StandardLevel;
26  
27  // a modified form of the LogEventAdapter in log4j2, which uses the log4j 1 LoggingEvent class
28  
29  /**
30   * Converts a Log4j 2 LogEvent into the components needed by a Log4j 1.x LoggingEvent.
31   * This class requires Log4j 2.
32   */
33  public class LogEventAdapter extends LoggingEvent {
34  
35      /** generated serialVersionUID */
36  	private static final long serialVersionUID = -1455395715714529922L;
37  
38  	public LogEventAdapter(LogEvent event) {
39      	super(event.getLoggerFqcn(), Category.getInstance(event.getLoggerName()), 
40      			event.getTimeMillis(), getLevel(event), event.getMessage(), event.getThreadName(), 
41      			event.getThrown() == null ? null : new ThrowableInformation(event.getThrown()),
42      			null, new LocationInfo(event.getThrown(), event.getLoggerFqcn()), event.getContextData().toMap()); // ?
43          // this.event = event;
44      }
45  
46      /**
47       * Return the level of this event. Use this form instead of directly
48       * accessing the <code>level</code> field.
49       */
50      public static Level getLevel(LogEvent event) {
51          switch (StandardLevel.getStandardLevel(event.getLevel().intLevel())) {
52              case TRACE:
53                  return Level.TRACE;
54              case DEBUG:
55                  return Level.DEBUG;
56              case INFO:
57                  return Level.INFO;
58              case WARN:
59                  return Level.WARN;
60              case FATAL:
61                  return Level.FATAL;
62              case OFF:
63                  return Level.OFF;
64              case ALL:
65                  return Level.ALL;
66              default:
67                  return Level.ERROR;
68          }
69      }
70  }
